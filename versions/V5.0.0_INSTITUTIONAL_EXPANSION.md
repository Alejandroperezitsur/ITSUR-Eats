  use(req: Request, res: Response, next: NextFunction) {
    // 1. Obtener tenant del host o par√°metro
    const tenantId = this.extractTenantFromHost(req.get('host'));
    
    // 2. Validar que tenant existe

async notifyWebhooks(event: string, data: any) {
  const webhooks = await this.webhooksRepository.find({
    where: { event, tenant_id: this.getTenant().id }
  });

  for (const webhook of webhooks) {
    try {
      await axios.post(webhook.url, {
        event,
        data,
        timestamp: new Date(),
        signature: this.generateSignature(data)
      }, {
        timeout: 5000,
        retries: 3
      });
    } catch (error) {
      // Guardar fallo para retry

        webhook_id: webhook.id,
        error: error.message
      });
    }
  }

@Injectable()
export class TenantGuard implements CanActivate {
  canActivate(context: ExecutionContext): boolean {
    const request = context.switchToHttp().getRequest();
    const user = request.user;
    const requestedTenant = request.tenant;

    // Usuario solo puede acceder a su tenant
